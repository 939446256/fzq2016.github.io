AVPlayerViewController

  1. 父类是UIViewController
  2. 需要设置成员变量AVPlayer，通过它实现播放功能
  3. iOS9.0之后, 需要使用AVPlayerViewController播放视频

AVPlayerViewController播放远程视频

  1. objc

     // 0. 准备工作
     // 01 导入框架
         #import <AVFoundation/AVFoundation.h>
         #import <AVKit/AVKit.h>
     // 02 iOS9.0 网络请求适配(HTTPS-->HTTP)需在info.plist配置NSAppTransportSecurity
     // 03 设置成员变量
     @property (nonatomic, readwrite, strong) AVPlayerViewController *playerVC;

 // 1. 根据URL创建AVPlayer
     NSURL *remoteURL = [NSURL URLWithString:@"http://v1.mukewang.com/57de8272-38a2-4cae-b734-ac55ab528aa8/L.mp4"];
     AVPlayer *player = [AVPlayer playerWithURL:remoteURL];

 // 2. 根据AVPlayer, 创建AVPlayerViewController控制器
         _playerVC = [[AVPlayerViewController alloc] init]; // 保证不被销毁并能重复利用
          _playerVC.player = player;

 // 3. 设置播放视图frame, 添加到需要展示的视图上
     // 3.1.1 设置播放视图的frame
     self.playerVC.view.frame = CGRectMake(0, 0, self.view.bounds.size.width, self.view.bounds.size.height * 9 / 16);

     // 3.1.2 添加播放视图到要显示的视图
     [self.view addSubview:self.playerVC.view];

     // 或3.2 弹出控制器
     // [self presentViewController:self.playerVC animated:YES completion:nil];*

 // 4.播放
     // 开始播放
     [self.playerVC.player play];

  2. swift

     // 0. 准备工作
     // 01 导入框架
         import AVFoundation
         import AVKit
     // 02 iOS9.0 网络请求适配(HTTPS-->HTTP)需在info.plist配置NSAppTransportSecurity

     // 1. 懒加载AVPlayerViewController
     lazy var movieVC: AVPlayerViewController? = AVPlayerViewController()

 override func viewDidLoad() {
     super.viewDidLoad()

     // 2. 根据URL创建AVPlayer
     let url = NSURL(string: "http://v1.mukewang.com/a45016f4-08d6-4277-abe6-bcfd5244c201/L.mp4")
     let player = AVPlayer(URL: url!)

     // 3. 设置player及属性（属性设置可选）
     movieVC?.player = player
     // 隐藏UI
//        movieVC?.showsPlaybackControls = false
     // 画中画
     movieVC?.allowsPictureInPicturePlayback = true
 }

 override func touchesBegan(touches: Set<UITouch>, withEvent event: UIEvent?) {

     // 容错处理：已经弹出控制器则不再弹出
     if presentedViewController != nil {
         return
     }

     // 4. 弹出控制器
     presentViewController(movieVC!, animated: true)         {
         // 5. 播放
         self.movieVC?.player?.play()
     }
 }
